using python 3.13.0

mkdir -p apps/server
cd apps/server/

create virtual environment 
python -m venv .venv

activate the virtual environment
source .venv/bin/activate

Install required packages 
pip install fastapi sqlalchemy uvicorn alembic asyncpg pydantic-settings

create a .gitignore file 
https://github.com/fastapi/fastapi/blob/master/.gitignore

freeze the dependencies
pip freeze > requirements.txt

Alembic setup
alembic init alembic

poetry setup 

poetry run black .

poetry run isort .

poetry install

poetry run uvicorn src.api.main:app --reload
